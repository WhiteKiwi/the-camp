	//CDN 업로드 파일 full url 세팅
	function fn_getFileSrcUrl(obj) {
		
		var cdnServerMode = "d";
		
		if(serverMode == "real") {
			cdnServerMode = "p";
		}
		else if(serverMode == "stg") {
			cdnServerMode = "s";
		}
		
		var sFileType = obj.fileTypeCd;
		var sImageThumbnailSize = '';
		
		if(obj.filePath == '') {
			return "";
		}
		
		var sFilePath = sCdnFileSrcUrl + obj.filePath + obj.savedFileNm;
		
		if(obj.extNm != null && obj.extNm != '' && typeof obj.extNm != 'undefined') {
			sFilePath += '.' + obj.extNm;
		}
			
		if(sFileType == sImageCommCd) {
			
			if(obj.thumb == null || obj.thumb == '' || typeof obj.thumb == 'undefined') {
				sImageThumbnailSize = thumbSizeMgr.defaultOriginSize;
			}
			else {
				sImageThumbnailSize = obj.thumb;
			}
			
			sFilePath += "?d=" + sImageThumbnailSize + "&s=" + cdnServerMode;
		}
		else {
			sFilePath += "?s=" + cdnServerMode;
		}
		
		return sFilePath;
	}

//캠프 미가입회원 SMS 발송
function fn_campInviteSms() {
	
	var broswerInfo = navigator.userAgent;

	if(broswerInfo.toLowerCase().indexOf("android")>-1 || broswerInfo.toLowerCase().indexOf("iphone")>-1 || broswerInfo.toLowerCase().indexOf("ipad")>-1){
		
		var phoneList = [];
		phoneList.push($("#invitePhoneNo").val());
		
		NativeCall("sendSms",{"number":phoneList,"message":$("#inviteSmsMessage").val()});
		
	} else {
		alert("더캠프 회원초대는 더캠프 앱에서만 가능합니다.");
	}
	
}

function fn_updateTalkContentLike(eventMgrSeq) {
	var ajaxOption = {
		url : "/event/content/insertTalkContentLikeA.do",
		data : {
			eventMgrSeq : eventMgrSeq
		},
		callbackFn : function(res) {
			if (res.resultCd == "0000") {
				if (res.flag == "insert") {
					$("#likeCnt"+eventMgrSeq).addClass("active").html(Number($("#likeCnt"+eventMgrSeq).html()) + 1);
				} else {
					$("#likeCnt"+eventMgrSeq).removeClass("active").html(Number($("#likeCnt"+eventMgrSeq).html()) - 1);
				}
			}
// 			alert(res.resultMsg);
		}
	}
	commonUtil.fnAjaxCall(ajaxOption);
}

function fn_cafeCreateCheck(regOrder, name,enterDate,birth,trainUnitTypeCd,trainUnitCd, grpCd, traineeRelationshipCd ) {

	var jsonData = {"regOrder":regOrder, "name":name, "enterDate":enterDate, "birth":birth, "trainUnitTypeCd":trainUnitTypeCd, "trainUnitCd":trainUnitCd, "grpCd":grpCd, "traineeRelationshipCd":traineeRelationshipCd};

		url: PAGE_CONTEXT + "/main/cafeCreateCheckA.do",
		data: jsonData,
		callbackFn: function(res) {
			//console.log(res);
			//alert(res.resultMsg);
			if(res.resultCd == "0000") {
				//alert("훈련소 메뉴에서 카페 가입을 하시기 바랍니다.");

			} else if(res.resultCd == "M001") {
				alert("카페 개설전.\n\n" +
						"1. \"오픈카페 전체보기\"에서 개설된 카페를 확인할 수 있습니다.\n\n" +

						"입영일부터 10일 이내에 개설됩니다.\n" +
